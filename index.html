Quick Start Commands 
# Clone the repository
git clone https://github.com/realmg51-cpu/NanoMind.git

# Navigate to directory
cd NanoMind

# Build and run with .NET
dotnet run

# Or run the downloaded executable
# Windows:
.\NanoMind-win-x64-v1.0.0.exe
# Linux/macOS:
chmod +x NanoMind-linux-x64-v1.0.0 && ./NanoMind-linux-x64-v1.0.0

# Or use Docker
docker build -t nanomind .
docker run --rm nanomind
